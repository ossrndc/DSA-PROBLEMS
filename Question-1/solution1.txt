#include <bits/stdc++.h>
using namespace std;
int main() {
    int t;
    cin>>t;
    while(t--){
        int n,a,ans=0,i;
        cin>>n;
        map<int,int>m;
        for(int i=0;i<n;i++){
            cin>>a;
            m[a]++;
        }
        for(auto i:m){
            ans+=(i.second/3);
        }
        cout<<ans<<endl;
    }

    return 0;
}
